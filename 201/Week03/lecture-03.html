## @knitr logismle
# logistic (-log(likelihood)) function with a covariate vector 'cv'
logislike=function(bet,dat,cv) {
  length(dat)*log(bet[3])  +2*sum(log(cosh((dat-bet[1]-bet[2]*cv) / (2*bet[3]))))}
z=2*rnorm(20) 
y=z+rlogis(20)
